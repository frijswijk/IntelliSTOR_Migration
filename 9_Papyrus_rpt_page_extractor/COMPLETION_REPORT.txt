PAPYRUS INTEGRATION:
shell_method extract_pages(input, rule, output_txt, output_pdf)
{
    declare local result = execute_command(
        "/path/to/python3 /path/to/script.py " &
        "%TMPFILE{input} " & rule & " " &
        "%TMPFILE{output_txt} " &
        "%TMPFILE{output_pdf}"
    );
    return result == 0 ? 0 : 1;
}
